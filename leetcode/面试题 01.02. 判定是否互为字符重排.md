[面试题 01.02. 判定是否互为字符重排](https://leetcode-cn.com/problems/check-permutation-lcci/)
```go
func CheckPermutation(s1 string, s2 string) bool {
    if len(s1) != len(s2) {
        return false
    }
    m1 := make(map[byte]int)
    for i := 0;i < len(s1);i++ {
            m1[s1[i]]++
            m1[s2[i]]--
    }
    for k := range m1 {
        if m1[k] != 0 {
            return false
        }
    }
    return true
}
```
